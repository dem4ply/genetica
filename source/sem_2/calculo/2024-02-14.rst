**************
funcion cubica
**************

.. math::
	ax^3 + bx^2 + cx + d

donde a, b, c y d son constantes; :math: `a \neq 0`

.. math::
	f(x) = x^3 

	Dom_x = \mathbb{R}

	Rango = \mathbb{R}

.. csv-table:: tabla
   :header: "x", "f( x )"
   :widths: 3, 10

	-3, -27
	-2, -8
	-1, -1
	0, 0
	1, 1
	2, 8
	3, 27

.. plot:: sem_2/calculo/plots/cubica.py

.. math::
	f(x) = x^3 + 1

	x^3 + 1 = 0
	\rightarrow x^3 = -1
	\rightarrow x = \sqrt[3]{-1}
	\rightarrow x = -1

	Dom_x = \mathbb{R}

	Rango = \mathbb{R}

.. plot:: sem_2/calculo/plots/cubica+1.py
